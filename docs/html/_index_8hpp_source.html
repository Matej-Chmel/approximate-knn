270 inline void Index<T>::fillHeap(
271 Heap<Comparator>& h, const Neighbors& N, const uint queryID,
272 const float* const latestData, const uint latestID
273 ) {
274 const auto data = this->space.getData(queryID);
275 h.clear();
276 h.push(this->space.getDistance(data, latestData), latestID);
277
278 if constexpr(T::usePrefetch) {
279 const auto lastIdx = N.len() - 1;
280 this->space.prefetch(N.get(0));
281
282 for(size_t i = 0; i < lastIdx; i++) {
283 this->space.prefetch(N.get(i + 1));
284
285 const auto& id = N.get(i);
286 h.push(this->space.getDistance(data, id), id);
287 }
288
289 const auto& id = N.get(lastIdx);
290 h.push(this->space.getDistance(data, id), id);
291
292 } else
293 for(const auto& id : N)
294 h.push(this->space.getDistance(data, id), id);
295 }

297 template<class T>
298 inline void Index<T>::insert(
299 const float* const queryData, const uint queryID
300 ) {
301 const auto L = this->entryLevel;
302 const auto l = this->gen.getNext();
303
304 this->conn.init(queryID, l);
305 this->resetEp(queryData);
306
307 for(auto lc = L; lc > l; lc--)
308 this->searchUpperLayer(queryData, lc);
309
310 for(auto lc = std::min(L, l);; lc--) {
311 this->searchLowerLayer<false>(queryData, this->cfg.efConstruction, lc, queryID);
312 const auto W = this->selectNewNeighbors(queryID, lc);
313
314 const auto mLayer = !lc ? this->cfg.mMax0 : this->cfg.mMax;
315
316 for(const auto& id : W) {
317 auto N = this->conn.getNeighbors(id, lc);
318
319 if(N.len() < mLayer)
320 N.push(queryID);
321 else
322 this->shrinkNeighbors(mLayer, id, N, queryData, queryID);
323 }
324
325 if(!lc)
326 break;
327 }
328
329 if(l > L) {
330 this->entryID = queryID;
331 this->entryLevel = l;
332 }
333 }

423 template<bool searching>
424 inline void Index<T>::searchLowerLayer(
425 const float* const query, const uint ef, const uint lc, const uint countBeforeQuery
426 ) {
427 this->heaps.prepareLowerSearch(this->ep);
428 this->visited.prepare(countBeforeQuery, this->ep.id);
429 auto& C = this->heaps.near;
430 auto& W = this->heaps.far;
431
432 while(C.len()) {
433 uint cand{};
434
435 {
436 const auto& c = C.top();
437 const auto& f = W.top();
438
439 if constexpr(searching) {
440 if(c.distance > f.distance)
441 break;
442 } else {
443 if(c.distance > f.distance && W.len() == ef)
444 break;
445 }
446
447 cand = c.id;
448 }
449
450 // Extract nearest from C.
451 C.pop();
452 const auto N = this->conn.getNeighbors(cand, lc);
453
454 if constexpr(T::usePrefetch) {
455 if constexpr(!Index<T>::useBitArray()) {
456 if(!N.len())
457 continue;
458
459 this->visited.prefetch(N.get(0));
460 }
461
462 VisitResult visRes = this->visited.insertNext(N, 0);
463
464 if(visRes.success)
465 this->space.prefetch(visRes.neighborID);
466
467 for(;;) {
468 if(!visRes.success)
469 break;
470
471 const auto currID = visRes.neighborID;
472
473 visRes = this->visited.insertNext(N, visRes.idx + 1);
474
475 if(visRes.success)
476 this->space.prefetch(visRes.neighborID);
477
478 this->processNeighbor(currID, query, ef, W);
479 }
480
481 } else
482 for(const auto& id : N)
483 if(this->visited.insert(id))
484 this->processNeighbor(id, query, ef, W);
485 }
486 }

489 inline void Index<T>::searchUpperLayer(
490 const float* const query, const uint lc
491 ) {
492 uint prev{};
493
494 do {
495 const auto N = this->conn.getNeighbors(this->ep.id, lc);
496 prev = this->ep.id;
497
498 if constexpr(T::usePrefetch) {
499 const auto len = N.len();
500
501 if(!len)
502 continue;
503
504 const auto lastIdx = len - 1;
505 this->space.prefetch(N.get(0));
506
507 for(size_t i = 0; i < lastIdx; i++) {
508 this->space.prefetch(N.get(i + 1));
509
510 const auto cand = N.get(i);
511 const auto distance = this->space.getDistance(query, cand);
512
513 if(distance < this->ep.distance) {
514 this->ep.distance = distance;
515 this->ep.id = cand;
516 }
517 }
518
519 const auto cand = N.get(lastIdx);
520 const auto distance = this->space.getDistance(query, cand);
521
522 if(distance < this->ep.distance) {
523 this->ep.distance = distance;
524 this->ep.id = cand;
525 }
526
527 } else {
528 for(const auto& cand : N) {
529 const auto distance = this->space.getDistance(query, cand);
530
531 if(distance < this->ep.distance) {
532 this->ep.distance = distance;
533 this->ep.id = cand;
534 }
535 }
536 }
537 } while(this->ep.id != prev);

551 template<class T>
552 inline Neighbors Index<T>::selectNewNeighborsHeuristic(Neighbors& R) {
553 if(this->heaps.far.len() <= this->cfg.mMax) {
554 R.fillFrom(this->heaps.far, this->ep);
555 return R;
556 }
557
558 this->heaps.prepareHeuristic();
559 auto& W = this->heaps.near;
560
561 {
562 const auto& e = W.top();
563 this->ep.distance = e.distance;
564 this->ep.id = e.id;
565 R.push(e.id);
566 }
567
568 W.pop();
569
570 while(W.len() && R.len() < this->cfg.mMax) {
571 {
572 const auto& e = W.top();
573 const auto eData = this->space.getData(e.id);
574
575 if constexpr(T::usePrefetch) {
576 const auto lastIdx = R.len() - 1;
577 this->space.prefetch(R.get(0));
578
579 for(size_t i = 0; i < lastIdx; i++) {
580 this->space.prefetch(R.get(i + 1));
581
582 if(this->space.getDistance(eData, R.get(i)) < e.distance)
583 goto isNotCloser;
584 }
585
586 if(this->space.getDistance(eData, R.get(lastIdx)) < e.distance)
587 goto isNotCloser;
588
589 } else {
590 for(const auto& rID : R)
591 if(this->space.getDistance(eData, rID) < e.distance)
592 goto isNotCloser;
593 }
594
595 R.push(e.id);
596
597 if(e.distance < this->ep.distance) {
598 this->ep.distance = e.distance;
599 this->ep.id = e.id;
600 }
601 }
602
603 isNotCloser:;
604
605 // Extract nearest from W.
606 W.pop();
607 }
608
609 return R;
610 }
