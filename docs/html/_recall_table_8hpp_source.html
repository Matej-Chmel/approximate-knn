264 inline void RecallTable<T>::run(

266 const nl::json& datasets, std::ostream& s

268 Timer timer{};
269 this->benchmarks.clear();
270
271 s << "Building index.\n";
272
273 timer.reset();
274 const typename Dataset<T>::Ptr dataset = Dataset<T>::getDataset(
275 configPath, dataDir, datasets, this->cfg.datasetName, s
276 );
277 auto index = dataset->getIndex(
278 this->cfg.efConstruction, this->cfg.mMax, this->cfg.seed, this->cfg.simdType
279 );
280 dataset->build(index);
281 this->buildElapsed = timer.getElapsed();
282
283 this->datasetStr = dataset->getString();
284 this->indexStr = index->getString();
285
286 s << "Index built in ";
287 prettyPrint(this->buildElapsed, s);
288 s << "\n\n";
289
290 for(const auto& efSearch : this->cfg.efSearchValues) {
291 auto& benchmark = this->benchmarks.emplace_back(efSearch);
292
293 s << "Querying with efSearch = " << efSearch << ".\n";
294
295 timer.reset();
296 const auto knnResults = dataset->query(index, efSearch);
297 benchmark.setElapsed(timer.getElapsed());
298
299 s << "Completed in ";
300 benchmark.prettyPrintElapsed(s);
301 s << "\nComputing recall.\n";
302
303 timer.reset();
304 benchmark.setRecall(dataset->getRecall(knnResults));
305 const auto recallElapsed = timer.getElapsed();
306 s << "Recall " << benchmark.getRecall() << " computed in ";
307 prettyPrint(recallElapsed, s);
308 s << "\n\n";
309 }
